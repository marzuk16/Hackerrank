Problem: Type of Triangle
    Solution: 
        SELECT
            CASE
                WHEN (A + B) <= C OR (B + C) <= A OR (A + C) <= B THEN 'Not A Triangle'
                ELSE
                    CASE
                        WHEN A = B AND B = C THEN 'Equilateral'
                        WHEN A = B OR B = C OR A = C THEN 'Isosceles'
                        WHEN A != B AND B != C THEN 'Scalene' 
                    END
            END
        FROM TRIANGLES;
